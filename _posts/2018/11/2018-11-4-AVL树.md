#include<bits/stdc++.h>
using namespace std;
typedef struct node {
	int val;
	node* left;
	node* right;

};

int getH(node* root) {
	if (root == NULL) return 0;
	int l = getH(root->left);
	int r = getH(root->right);
	return max(l,r)+1;
}

node* leftroll(node* root) {
	node* t = root->right;
	root->right = t->left;
	t->left = root;
	return t;
}

node* rightroll(node* root) {
	node* t;
	t = root->left;
	root->left = t->right;
	t->right = root;
	return t;
}

node* leftrightroll(node* root) {
	root->left = leftroll(root->left);
	return rightroll(root);
}

node* rightleftroll(node* root) {
	root->right = rightroll(root->right);
	return leftroll(root);
}

node* insert(int t,node *root) {
	if (root == NULL) {
		root = new node();
		root->val = t;
	}
	else if (root->val > t) {
		root->left=insert(t, root->left);
		int l = getH(root->left);
		int r = getH(root->right);
		if (l - r >= 2) {
			if (t < root->left->val) {
				root = rightroll(root);
			}
			else {
				root = leftrightroll(root);
			}
		}
	}
	else {
		root->right=insert(t, root->right);
		int l = getH(root->left);
		int r = getH(root->right);
		if (r - l >= 2) {
			if (t > root->right->val) {
				root = leftroll(root);
			}
			else {
				root = rightleftroll(root);
			}
		}
	}
	return root;
}
int is = 1, af = 0;
vector<int> levelorder(node *root) {
	vector<int> v;
	queue<node*> q;
	q.push(root);
	while (!q.empty()) {
		node* t = q.front();
		q.pop();
		v.push_back(t->val);
		if (t->left != NULL) {
			if (af) is = 0;
			q.push(t->left);
		}
		else {
			af = 1;
		}
		if (t->right != NULL) {
			if (af) is = 0;
			q.push(t->right);
		}
		else {
			af = 1;
		}
	}
	return v;
}
